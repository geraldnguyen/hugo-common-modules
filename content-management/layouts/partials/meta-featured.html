{{/* meta-featured.html Parameters (passed as a dict): Page -> the current page/context (required) Meta -> taxonomy key (must exist in Site.Taxonomies) (required) Terms -> slice of taxonomy term names or a single term string (required) Random -> integer Y (>=0). Picks Y random pages from eligible non-featured pages. Returns: featured pages from eligible pages, plus optional random non-featured pages. */}} {{ $ctx := . }} {{ $page := $ctx.Page }} {{ if not (isset $ctx "Meta") }} {{ errorf "meta-featured: missing required parameter 'Meta'" }} {{ end }} {{ if not (isset $ctx "Terms") }} {{ errorf "meta-featured: missing required parameter 'Terms'" }} {{ end }} {{ $metaKey := $ctx.Meta }} {{ $termsParam := $ctx.Terms }} {{ $random := cond (isset $ctx "Random") $ctx.Random 0 }} {{/* Meta must point to an existing taxonomy */}} {{ $taxonomy := index $page.Site.Taxonomies $metaKey }} {{ if not $taxonomy }} {{ errorf "meta-featured: Meta '%s' is not a configured taxonomy" $metaKey }} {{ end }} {{/* Normalize Terms to a slice */}} {{ $terms := slice }} {{ if eq (printf "%T" $termsParam) "string" }} {{ if ne $termsParam "" }} {{ $terms = slice $termsParam }} {{ end }} {{ else }} {{ $terms = $termsParam }} {{ end }} {{ if eq (len $terms) 0 }} {{ errorf "meta-featured: Terms must be a non-empty string or slice" }} {{ end }} {{/* Extract taxonomy terms whose value type is page.TaxonomyList */}} {{ $eligible := slice }} {{ range $taxTerm, $taxValue := $taxonomy }} {{ range $term := $terms }} {{ $inputTerm := lower (printf "%v" $term) }} {{ if eq $inputTerm (lower $taxTerm) }} {{ $eligible = $eligible | append $taxValue }} {{ end }} {{ end }} {{ end }} {{/* Flatten eligible pages (in case of multiple matching terms) and remove duplicates */}} {{ $eligible = $eligible | uniq }} {{/* Select featured pages from eligible set */}} {{ $featured := where $eligible "Params.featured" true }} {{ $selected := $featured }} {{/* Optional random additions from eligible pages, excluding already-selected featured pages */}} {{ if gt $random 0 }} {{ $featuredPaths := slice }} {{ range $featured }} {{ $featuredPaths = $featuredPaths | append .Path }} {{ end }} {{ $remainder := slice }} {{ range $eligible }} {{ if not (in $featuredPaths .Path) }} {{ $remainder = $remainder | append . }} {{ end }} {{ end }} {{ $more := (shuffle $remainder) | first $random }} {{ range $more }} {{ $selected = $selected | append . }} {{ end }} {{ end }} {{ return $selected }}
